def check_lipinski_rule(molecula):
    if ((Lipinksy.NumHDonors(molecula) <= 5) and
        (Lipinksy.NumHAcceptors(molecula) <= 10) and
        (Lipinksy.rdMolDescriptors.CalcExactMolWt(molecula) < 500) and
        (Lipinksy.rdMolDescriptors.CalcCrippenDescriptors(molecula)[0] <= 5)):
        return True
    else:
        return False
    
final_candidate = []

    except:
        pass
