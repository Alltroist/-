def generate_zmatrix(bond_length=1.52986, valence_angle=111.200):
    #    ""     ORCA
    #        
    #    4  -    (C-C).
    # 5  -     HC
    # 6  -    CC
    inp = '''!HF RHF 6-31G
    * int 0 1
    C 0 0 0 0 0 0 
    C 1 0 0 %s 0 0 
    H 1 2 0 1.08439 %s 0
    H 1 2 3 1.08439 %s 120
    H 1 2 3 1.08439 %s -120
    H 2 1 3 1.08439 %s 180
    H 2 1 5 1.08439 %s 120
    H 2 1 5 1.08439 %s -120
    *
    ''' % (bond_length, valence_angle, valence_angle, valence_angle, valence_angle, 
           valence_angle, valence_angle)
    return inp

def generate_zmatrix_tors(torsion_angle):
    #      
    inp = '''!HF RHF 6-31G
    * int 0 1
    C 0 0 0 0 0 0 
    C 1 0 0 1.52986 0 0 
    H 1 2 0 1.08439 111.200 0
    H 1 2 3 1.08439 111.200 120
    H 1 2 3 1.08439 111.200 -120
    H 2 1 3 1.08439 111.200 %s
    H 2 1 5 1.08439 111.200 %s
    H 2 1 5 1.08439 111.200 %s
    *
    ''' % (torsion_angle, torsion_angle - 60, torsion_angle + 60)
    return inp

def fit_and_plot(x, y, fit_type):
    
    if fit_type == "polinomial":
        fitfunc = lambda p, x: p[0] + p[1]*x + p[2]*x*x + p[3]/x + p[4]/x/x
        p_0 = np.zeros(5)
    elif fit_type == "trigonometric":
        fitfunc = lambda p, x: p[0] + (p[1] + p[2]*x) * np.sin(p[3] + p[4]*x) + (p[5] + p[6]*x) * np.cos(p[7] + p[8]*x) 
        p_0 = np.ones(9)
    elif fit_type == "exponential":
        fitfunc = lambda p, x: p[0] + (p[1] + p[2]*x) * np.exp(p[3] + p[4]*x)
        p_0 = np.ones(5)
    else:
        print "Invalid fit_type"
        exit(1)
        
    errfunc = lambda p, x, y: fitfunc(p, x) - y # Error function

    p1, success = optimize.leastsq(errfunc, p_0, args=(x, y))
    print "Optimized params:", p1
    print "Function error = ", np.linalg.norm(errfunc(p1, x, y))

    #Plot it
    plt.plot(x, y, "ro", x, fitfunc(p1, x),"r-", c='blue', alpha=0.5)
    step = (x[1] - x[0])
    plt.xlim(x[0] - 5*step, x[-1] + 5*step)
    plt.show()
